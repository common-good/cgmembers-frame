function ray($a) {return explode(' ', $a);}

function opts($ray, $dft = '') {
  $res = '';
  foreach ($ray as $k => $v) {
    $sel = $k == $dft ? ' SELECTED' : '';
    $res .= "<option value=\"$k\"$sel>$v</option>\n";
  }
  return $res;
}
